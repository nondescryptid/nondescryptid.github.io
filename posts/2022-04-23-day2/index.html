def timeConversion(s):
    # Write your code
    if s[-2:] == "PM":
        s = s.replace("PM", "")
        digits = int(s[0:2])
        if digits != 12:
            digits += 12
            digits = str(digits)
            s = digits + s[2:]
        else:
            pass

    else:
        s = s.replace("AM", "")
        if s[0:2] == "12":
            s = "00" + s[2:]
    return s
